1. ------------kaj je z spi1 ?
   1.a) spi mora biti omogočen (rasp-config, advanced, ...)
   1.b) preveri   
   # ls -al /dev/spi*
    
	/dev/spidev0.0  /dev/spidev0.1  /dev/spidev1.0  /dev/spidev1.1  /dev/spidev1.2
	
	(vse so v sintaksi bus.device oz. port.device  )
	
	rabimo zadnjega (po predvidenem setup-u), če ga ni:
	dodaš v /boot/config.txt vrstico
	   dtoverlay = spi1-3cs
	in rebootaš pi-ja. 3cs mora biti, ker 1cs, 2cs omogočita zgolj prvi ali rva dva device-a
	
2.setup (rpi<-->display) je:

     gpio            oznake na mojem displayu
	 ---------------------------------------------
     gpio 16 ce2	/CS
     gpio 20 mosi	/SDA
     gpio 21 clk	/SCL	
	 
	 gpio 26 		/RESET
	 gpio 6 dc		/A0
	 
	 referenca a : https://pinout.xyz/pinout/spi
	 
	 16,20,21 predpisuje spi1, za reset,dc uporabiš poljubna neuporabljena porta (6, 26)

import time

from luma.core.interface.serial import  spi 
import luma.core.interface.parallel as parallel
from luma.core.render import canvas
from luma.lcd.device import  st7735


#serial = spi(port=0, device=1, gpio_DC=23, gpio_RST=24)
serial = spi(port=1, device=2, gpio_DC=6, gpio_RST=26)      # spidev1.2  (CE02)
device = st7735(serial)


with canvas(device) as draw:
    draw.rectangle(device.bounding_box, outline="white", fill="black")  # outline=(r,g,b)
    draw.text((30, 40), "Hello World", fill="red")                      # fill=(r,g,b)

time.sleep(10)
