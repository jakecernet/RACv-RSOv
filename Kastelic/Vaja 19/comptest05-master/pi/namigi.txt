---------------------------- orodja  so v /sbin : iw*, wpa_*

// -- kako do imen interface-ov ?
iw dev            # wireless
ip a, ip list     # vsi interf
ip a show wlan0   # test configuracije wlan0  (ali ima ip ? )

wlan0   -- default wifi

iw wlan0 scan  # vrn scan omrežja po access pointih, če rabiš samo SSID-je:
iw wlan0 scan | grep SSID

iw wlan0 link   # a je kaj z njim : a je povezan ? na kateri SSID? s kakšno hitrostjo

------------------------------------------------------------------------- postopkek
(alternativa je tule : https://www.raspberrypi.org/documentation/configuration/wireless/wireless-cli.md)
postopke postavitve s konzole :  (verjetno s SUDO :)

iw wlan0 scan  # najdemo SSID - recimo mojSSID
wpa_passphrase mojSSID >> /etc/wpa_supplicant.conf              #generiramo koniguaracijo v /etc/wpa_supplicant.conf  - pozove na vpis passworda
                                                                #preveri supplicat : cat /etc/wpa_supplicant.conf
wpa_supplicant -B -D wext -i wlan0 -c /etc/wpa_supplicant.conf  #connect to wireless : -D wext je driver (-Dnnl80211,wext)
#wpa_supplicant -B -i wlan0  -c /etc/wpa_supplicant.conf
#wpa_supplicant -i wlan0 -c /etc/wpa_supplicant/wpa_supplicant.conf -B -D   #hm, če driverja ne specificiraš, prevzame default linux driver
iw wlan0 link                                                   #testiramo
dhclient wlan0                                                  #ta se verjetno ne bo autoatično zagnal : get ip from DHCP
