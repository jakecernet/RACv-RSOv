##### tr #####

# echo koza stara       - izpiše niz koza stara na zaslon (stdout) 
echo koza stara | tr 'ak' '12'    # zamenja vse a-je z 1 in se k-je z 2 , izpiše na standardni izhod
echo koza stara | tr -d 'a'       # odstrani vse a-je iz niza; pomni -d !

echo '  - 		:-koza stara ' | tr -d '[:space:]'     #odstrani vse neizpisljive znake iz niza (presledke,tabulatorje, ...)

#man tr da še nekaj uporabnih vzorcev [:digit:] ali [:lower:]

echo 'Sosedova krava je nora' | tr [:lower:] [:upper:]  # niz v uppercase ....


##### cut #####

#pri naslednih testih bo vhod ls -al na -home- folderju (/home/pi)
#ls -al da izhod v obliki :
# atributi nodes user group dolžina mesec dan ura ime    ---> 9 polj, ločenih s presledki

###### -f : fields v kombinaciji z -d : delimiter med polji
ls -al | cut -d ' ' -f2    # izpiše drugo polje (nodes)
ls -al | cut -d ' ' -f1,3  # izpiše prvo in tretje polje; polja ločena s presledkom

## zadeva pri ls -al ni zanesljiva, ker polja niso vedno locena s po enim presledkom, če to želiš, rabiš več procesiranja :

ls -al | sed -e 's/ \+/ /g'                 # pazi na presledke : sed - strream editor, uporabimo search in replace global (vse v vrstici);
                                            # za s - kot serach pride dve polji ločeni z /1/2/  v 1 je search kriterij, v 2 replace
                                            # / \+/ je regularen izraz, ki pravi, da zamenja enega ali več presledkov							
ls -al | sed -e 's/[[:space:]]\+/ /g'       # enako kot predhodni, le da zamenja vse white-space znake (posix)

#če rabimo odstraniti vse presledke
ls -al | sed 's/[[:space:]]//g'

# ker vemo, da so atributi v 1 stolpcu, imena datotek pa v 9-tem, sedaj lahko :
ls -al | sed -e 's/[[:space:]]\+/ /g' | cut -d' ' -f1,9
ali pa
ls -l | awk '{print $1,$9}'                 #  parmeter awk-ja je program, z uakzi, ki jih lahko nanizaš $1,$9 sta stolpca, ki jih kot spremeljivke okolja skripte
                                            #  uporabiš v programu, privzeti delimiter za polja oz. parametre poslane v awk program je presledek, lahko ga pa nastaviš
                                            #  z awk -F ':'

#### ce bi slo tole, bi bilo super pa ne gre .... :  echo 30.3 | awk 'printf(\"use remoteDB; insert into testtable values('','','$1');\")'   											

###### -c : po odmiku znakov od začetka niza

#dostopnostni atributi drwxrwxrwx    d - tip datoteke (direktorij, običajni file, character oriented, block oriented, temporary, ...) tole je direktorij, če ni nič je regular file
  rwx je v treh skupinah po 3 prva skupina pripada lastniku (user), druga skupini (group) tretja velja za vse ostale (ohers)
  
zanimajo nas zgolj atributi za lastnika: iz ls -al vidimo, da so to znaki 2,3,4 oz začetka vrstice :

ls -al | cut -c2,3,4     # vrne znake 2-4 od začetka vrstice -al
ls -al | cut -c2-4       # vrne znake 2-4 od začetka vrstice
ls -al | cut -c2-4,8-10  # vrne znake 2-4 in 8-10 od začetka vrstice

#!/bin/bash

# poskušamo sestaviti query za insert in nasloviti mysql strežnik
# v skriptah pri prirejanju vrednosti spremenljivkam ne pozabi, da pred ali za enačajem ne sme bit presledkov !

VALU=$(($RANDOM%500))
sql_string="use remoteDB; insert into testtable values('','',$VALU);"  

echo $sql_string | /usr/bin/mysql -u pepi -h 192.168.1.75 --password=pepipepi remoteDB # string direktno pošljemo klientu da da pošlje serverju

#če datum/timestamp ni defaault - current-timestamp, zgeneriramo in pošlemo še tega:
#time_st=`date +"%s"`
#meni timestamp ne deluje, karkoli zapišem je 0000-00-00 00:00:00 !!!! ampak to je baje MariaDB
#gremo raje z običajnim datumom, ki se sam pretvori v tiestamp ... (????)

time_st=`date +"%Y-%m-%d %H:%M:%S"`
sql_string1="use remoteDB; insert into testtable values('','$time_st',$VALU);"  
echo $sql_string1 | /usr/bin/mysql -u pepi -h 192.168.1.75 --password=pepipepi remoteDB
